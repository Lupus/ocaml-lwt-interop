1//! This module provides an implementation of a `Future` called `MlBoxFuture`,
2//! which bridges OCaml's Lwt futures with Rust's async ecosystem.
3//!
4//! # Overview
5//!
6//! `MlBoxFuture` is designed to integrate OCaml's Lwt futures with Rust's
7//! `async`/`await` syntax.
8//! Due to limitations in passing Rust callbacks directly to OCaml functions
9//! (e.g., it's not possible to call `Lwt.on_any` from Rust and pass a Rust
10//! closure), the approach is inverted:
11//!
12//! - On the OCaml side, Lwt futures are wrapped into `MlBoxFuture` via C stubs.
13//! - OCaml registers callbacks on the Lwt future to resolve or reject the
14//!   `MlBoxFuture` when the Lwt future completes.
15//! - This allows the Rust `MlBoxFuture` to be driven by the OCaml runtime,
16//!   enabling Rust code to `await` on Lwt futures.
17//!
18//! # Implementation Details
19//!
20//! The `MlBoxFuture` is largely based on the timer future example from the Rust
21//! `async` book:
22//! [Wakeups](https://rust-lang.github.io/async-book/02_execution/03_wakeups.html).
23//!
24//! It uses an `Arc<Mutex<SharedState>>` to manage shared state between the
25//! future and the code that will resolve or reject it.
26//! When the future is polled, it checks if the value is available. If not, it
27//! stores the waker to be woken up later.
28//! When the OCaml side resolves or rejects the future, it sets the value and
29//! wakes up the waker, allowing the future to complete.
30
31use std::{
32    future::Future,
33    pin::Pin,
34    sync::{Arc, Mutex},
35    task::{Context, Poll, Waker},
36};
37
38use ocaml_rs_smartptr::ml_box::MlBox;
39
40/// Shared state between the `MlBoxFuture` and the code that resolves or rejects
41/// it.
42///
43/// This struct holds the result value (once it's available), the waker to wake
44/// up the future when it's ready, and a flag indicating whether the future has
45/// been completed.
46#[derive(Debug)]
47struct MlBoxFutureSharedState {
48    /// The result value of the future, set when the future is resolved or
49    /// rejected.
50    value: Option<Result<MlBox, crate::error::Error>>,
51    /// The waker to notify the executor when the future is ready.
52    waker: Option<Waker>,
53    /// Indicates whether the future has been completed.
54    completed: bool,
55}
56
57/// A future that represents a computation which will eventually produce an
58/// `MlBox`.
59///
60/// `MlBoxFuture` is used to bridge OCaml's Lwt futures with Rust's async code.
61/// It can be awaited in Rust, and is resolved or rejected from the OCaml side.
62#[derive(Clone, Debug)]
63pub struct MlBoxFuture {
64    /// Shared state between the future and the resolver.
65    shared_state: Arc<Mutex<MlBoxFutureSharedState>>,
66}
67
68impl Future for MlBoxFuture {
69    type Output = Result<MlBox, crate::error::Error>;
70
71    /// Polls the future to check if it has been resolved or rejected.
72    ///
73    /// If the value is available, returns `Poll::Ready` with the result.
74    /// Otherwise, stores the waker and returns `Poll::Pending`.
75    fn poll(self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Self::Output> {
76        let mut shared_state = self.shared_state.lock().unwrap();
77        match shared_state.value.take() {
78            Some(maybe_value) => Poll::Ready(maybe_value),
79            None => {
80                // Store the waker so we can wake this task when the value
81                // is set.
82                shared_state.waker = Some(cx.waker().clone());
83                Poll::Pending
84            }
85        }
86    }
87}
88
89impl MlBoxFuture {
90    /// Creates a new `MlBoxFuture`.
91    ///
92    /// This future can later be resolved or rejected using the `resolve` or
93    /// `reject` methods.
94    pub fn new() -> Self {
95        let shared_state = Arc::new(Mutex::new(MlBoxFutureSharedState {
96            value: None,
97            waker: None,
98            completed: false,
99        }));
100        MlBoxFuture { shared_state }
101    }
102
103    /// Sets the value of the future and wakes up the waker if it's stored.
104    ///
105    /// This method is called internally by `resolve` and `reject`.
106    ///
107    /// # Panics
108    ///
109    /// Panics if the future has already been completed.
110    fn set_value(&self, value: Result<MlBox, crate::error::Error>) {
111        let mut shared_state = self.shared_state.lock().unwrap();
112        if shared_state.completed {
113            panic!("Attempt to resolve an already resolved promise")
114        }
115        shared_state.completed = true;
116        shared_state.value = Some(value);
117        // Wake up the waker to notify that the future is ready.
118        if let Some(waker) = shared_state.waker.take() {
119            waker.wake()
120        }
121    }
122
123    /// Resolves the future with the given `MlBox` value.
124    ///
125    /// Typically called from the OCaml side when the Lwt future is fulfilled.
126    pub fn resolve(&self, value: MlBox) {
127        self.set_value(Ok(value))
128    }
129
130    /// Rejects the future with the given error message.
131    ///
132    /// Typically called from the OCaml side when the Lwt future is rejected.
133    pub fn reject(&self, msg: String) {
134        self.set_value(Err(crate::error::Error::LwtPromiseRejection(msg)))
135    }
136}
137
138impl Default for MlBoxFuture {
139    fn default() -> Self {
140        Self::new()
141    }
142}
