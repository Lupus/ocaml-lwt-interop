1//! This module provides some low-level primities for managing OCaml domain
2//! lock. Tested on OCaml 4, should work on OCaml 5 just fine.
3//!
4//! # OCaml 5 Runtime System Acquisition
5//
6//! OCaml 5 introduced true parallelism through domains, which map to OS threads
7//! and allow OCaml code to run simultaneously on multiple CPU cores. However,
8//! this new model raises questions about how existing C code and thread pools
9//! interact with OCaml domains. Specifically, there is uncertainty around which
10//! OCaml domain will be used when C code calls functions like
11//! `caml_acquire_runtime_system`` to interact with the OCaml runtime. This is
12//! particularly relevant for scenarios where C thread pools are created from
13//! OCaml and need to execute OCaml code.
14//!
15//! ## Runtime System Functions
16//!
17//! In OCaml 5, `caml_acquire_runtime_system` and `caml_release_runtime_system`
18//! operate on a per-domain basis, acquiring or releasing the lock for the
19//! current domain associated with the calling OS thread[1]. These functions
20//! ensure thread-safety when C code interacts with the OCaml runtime. For
21//! backwards compatibility, they are still used in OCaml 5, but newer functions
22//! like `caml_domain_lock` and `caml_domain_unlock` are recommended for new
23//! code as they are more explicit about operating on the current domain[1].
24//! It's crucial to maintain proper balance between acquire and release calls
25//! within each execution path in C code to prevent deadlocks or other
26//! synchronization issues.
27//!
28//! ## C Thread Registration
29//!
30//! When creating C thread pools that need to interact with the OCaml runtime,
31//! each worker thread must call `caml_c_thread_register` to associate itself
32//! with the OCaml runtime system[2]. This registration process does not create
33//! a new OCaml domain, but rather allows the C thread to safely interact with
34//! existing domains. After registration, when these threads call
35//! `caml_acquire_runtime_system`, they will always acquire the lock for domain
36//! 0 (the main domain)[1]. This means all registered C threads compete for the
37//! same domain lock, potentially limiting parallelism in scenarios with
38//! multiple OCaml domains.
39//!
40//! ## Domain and Thread Interaction
41//!
42//! When a C thread pool is created from OCaml via a C stub, and worker threads
43//! call `caml_c_thread_register`, they become associated with the OCaml runtime
44//! but not with any specific domain. Subsequent calls to
45//! `caml_acquire_runtime_system` or `caml_domain_lock` from these threads will
46//! always acquire the lock for domain 0 (the main domain)[1]. This design
47//! choice means that all worker threads compete for the same domain lock,
48//! potentially limiting parallelism in multi-domain scenarios[1]. The
49//! association between a C thread and an OCaml domain is determined once at
50//! thread creation, and currently, the only way to create threads on non-zero
51//! domains is from OCaml code[1][3].
52//!
53//! ## References
54//!
55//! [1] <https://discuss.ocaml.org/t/test-caml-state-and-conditionally-caml-acquire-runtime-system-good-or-bad/12489>
56//! [2] <https://dev.to/yawaramin/practical-ocaml-multicore-edition-3gf2>
57//! [3] <https://ocaml.org/manual/5.2/parallelism.html>

59use std::{
60    ffi::c_int,
61    panic::{catch_unwind, UnwindSafe},
62    process::abort,
63};

65extern "C" {
66    /*
67    caml_acquire_runtime_system() The calling thread re-acquires the master lock
68    and other Caml resources. It may block until no other thread uses the Caml
69    run-time system.
70    */
71    // fn caml_acquire_runtime_system(); // For some reason this symbol is missing...
72
73    /*
74    caml_release_runtime_system() The calling thread releases the master lock
75    and other Caml resources, enabling other threads to run Caml code in
76    parallel with the execution of the calling thread.
77    */
78    // fn caml_release_runtime_system(); // For some reason this symbol is missing...
79
80    /* caml_enter_blocking_section as an alias for caml_release_runtime_system */
81    fn caml_enter_blocking_section();
82
83    /* caml_leave_blocking_section as an alias for caml_acquire_runtime_system */
84    fn caml_leave_blocking_section();
85
86    /*
87    caml_c_thread_register() registers the calling thread with the Caml run-time
88    system. Returns 1 on success, 0 on error. Registering an already-register
89    thread does nothing and returns 0.
90    */
91    fn caml_c_thread_register() -> c_int;
92
93    /*
94    caml_c_thread_unregister() must be called before the thread terminates, to
95    unregister it from the Caml run-time system. Returns 1 on success, 0 on
96    error. If the calling thread was not previously registered, does nothing and
97    returns 0.
98    */
99    fn caml_c_thread_unregister() -> c_int;
100}
101
102unsafe fn caml_acquire_runtime_system() {
103    caml_leave_blocking_section();
104}
105
106unsafe fn caml_release_runtime_system() {
107    caml_enter_blocking_section();
108}
109
110/// Registeres current thread with OCaml runtime, aborts the program if
111/// registration failed
112pub(crate) fn register_thread() {
113    if unsafe { caml_c_thread_register() } != 1 {
114        eprintln!("caml_c_thread_register() failed!");
115        abort()
116    }
117}
118
119/// Un-registeres current thread with OCaml runtime, aborts the program if
120/// un-registration failed
121pub(crate) fn unregister_thread() {
122    if unsafe { caml_c_thread_unregister() } != 1 {
123        eprintln!("caml_c_thread_unregister() failed!");
124        abort()
125    }
126}

128/// Runs `f` with OCaml domain lock being released. `f` **MUST NOT** use any
129/// functions that require OCaml domain lock to be held.
130pub(crate) fn with_released_lock<F, R>(f: F) -> R
131where
132    F: FnOnce() -> R + UnwindSafe,
133{
134    unsafe { caml_release_runtime_system() };
135    let result = catch_unwind(f);
136    unsafe { caml_acquire_runtime_system() };
137    match result {
138        Ok(value) => value,
139        Err(panic) => std::panic::resume_unwind(panic),
140    }
141}
142
143/// Runs `f` with OCaml domain lock being acquired. `f` can safely use any
144/// functions that require OCaml domain lock to be held.
145pub(crate) fn with_acquired_lock<F, R>(f: F) -> R
146where
147    F: FnOnce(&ocaml::Runtime) -> R + UnwindSafe,
148{
149    unsafe { caml_acquire_runtime_system() };
150    let gc = unsafe { ocaml::Runtime::recover_handle() };
151    let result = catch_unwind(|| f(gc));
152    unsafe { caml_release_runtime_system() };
153    match result {
154        Ok(value) => value,
155        Err(panic) => std::panic::resume_unwind(panic),
156    }
